```python
import pandas as pd
from datetime import datetime

def generate_response(row):
    """Generates a personalized response based on existing notes."""
    name = row.get('name', 'there')
    notes = str(row.get('notes', ''))
    
    if pd.notna(row.get('notes')) and notes.strip():
        return f"Hi {name}, it's been a while since we last spoke. I was reviewing my notes and saw: '{notes}'. I'd love to catch up and see how things are going!"
    else:
        return f"Hi {name}, it's been a few months since we last connected. I'd love to grab a quick coffee or call to see what you're up to lately!"

def process_leads(input_file, output_file):
    # Load the CSV
    try:
        df = pd.read_csv(input_file)
    except FileNotFoundError:
        print(f"Error: {input_file} not found.")
        return

    # Ensure required columns exist
    required_columns = {'last_contact'}
    if not required_columns.issubset(df.columns):
        print(f"Error: CSV must contain the following columns: {required_columns}")
        return

    # Convert last_contact to datetime objects
    df['last_contact'] = pd.to_datetime(df['last_contact'], errors='coerce')

    # Calculate the date 4 months ago from today
    four_months_ago = pd.Timestamp.now() - pd.DateOffset(months=4)

    # Filter leads where last_contact is older than 4 months
    # (i.e., last_contact date is less than the date 4 months ago)
    filtered_df = df[df['last_contact'] < four_months_ago].copy()

    # Add the personalized response column
    filtered_df['personalized_response'] = filtered_df.apply(generate_response, axis=1)

    # Export to the new CSV
    filtered_df.to_csv(output_file, index=False)
    print(f"Successfully processed leads. Output saved to {output_file}")

if __name__ == "__main__":
    # Example usage:
    # Ensure you have an 'input_leads.csv' with columns like: name, email, last_contact, notes
    INPUT_CSV = "input_leads.csv"
    OUTPUT_CSV = "filtered_leads_output.csv"
    
    process_leads(INPUT_CSV, OUTPUT_CSV)
```

import os
import csv
import json
from datetime import datetime
import google.generativeai as genai

# Configure Gemini API
genai.configure(api_key=os.environ.get("GEMINI_API_KEY"))
model = genai.GenerativeModel('gemini-1.5-flash')

def process_leads(input_file, output_file):
    today = datetime.now().strftime("%Y-%m-%d")
    
    try:
        with open(input_file, mode='r', encoding='utf-8') as infile:
            reader = csv.DictReader(infile)
            # Add the new column for the output CSV
            fieldnames = reader.fieldnames + ['personalized_response']
            
            with open(output_file, mode='w', encoding='utf-8', newline='') as outfile:
                writer = csv.DictWriter(outfile, fieldnames=fieldnames)
                writer.writeheader()
                
                for row in reader:
                    # Construct prompt for Gemini to evaluate and generate response
                    prompt = f"""
Today's date is {today}.
Here is a row of data representing a sales lead (columns might be sparse or arbitrary):
{row}

Task:
1. Figure out if this lead was last contacted strictly MORE than 4 months ago.
2. If yes, write a short, friendly 2-3 sentence personalized re-engagement message asking for a quick call or coffee. Use any available notes. If there are no notes or they are empty, default to: "Hi [Name/there], it's been a few months since we last connected. I'd love to grab a quick coffee or call to see what you're up to lately!"

Return ONLY a valid JSON object in the exact following format, with no markdown or extra text:
{{
    "older_than_4_months": true/false,
    "response": "Your generated response here, or an empty string if older_than_4_months is false"
}}
"""
                    try:
                        result = model.generate_content(prompt)
                        text = result.text.strip()
                        
                        # Strip markdown codeblocks if Gemini includes them
                        if text.startswith("```json"):
                            text = text[7:-3].strip()
                        elif text.startswith("```"):
                            text = text[3:-3].strip()
                            
                        data = json.loads(text)
                        
                        if data.get("older_than_4_months"):
                            row['personalized_response'] = data.get("response", "")
                            writer.writerow(row)
                            print(f"Kept lead: {row.get('name', 'Unknown')}")
                        else:
                            print(f"Skipped lead: {row.get('name', 'Unknown')} (recent contact)")
                            
                    except Exception as e:
                        print(f"Error processing row {row}: {e}\nResponse text was: {result.text if 'result' in locals() else 'N/A'}")

        print(f"Successfully processed leads. Output saved to {output_file}")
    except FileNotFoundError:
        print(f"Error: {input_file} not found.")

if __name__ == "__main__":
    INPUT_CSV = "input_leads.csv"
    OUTPUT_CSV = "filtered_leads_output.csv"
